class BaseScript(ABC):
    """
    Base class for all internal scripts
    
    Provides common functionality and structure for script templates
    """
    
    def __init__(self, name: str, description: str = "", version: str = "1.0"):
        self.name = name
        self.description = description
        self.version = version
        self.is_running = False
        self.start_time = None
        self.iteration_count = 0
        
        # Script metadata
        self.metadata = {
            'name': name,
            'description': description,
            'version': version,
            'author': 'SafeRS',
            'category': 'general',
            'tags': [],
            'requirements': []
        }
    
    @abstractmethod
    def setup(self) -> bool:
        """
        Setup method called before script execution
        Override this to initialize script-specific resources
        
        Returns:
            bool: True if setup successful, False otherwise
        """
        pass
    
    @abstractmethod
    def execute_iteration(self) -> bool:
        """
        Execute one iteration of the script
        Override this with your main script logic
        
        Returns:
            bool: True to continue, False to stop
        """
        pass
    
    @abstractmethod
    def cleanup(self) -> None:
        """
        Cleanup method called after script execution
        Override this to clean up script-specific resources
        """
        pass
    
    def run(self, duration: float = 60.0, callback: Optional[Callable] = None) -> bool:
        """
        Run the script for specified duration
        
        Args:
            duration: Duration in seconds to run the script
            callback: Optional callback for status updates
            
        Returns:
            bool: True if script completed successfully
        """
        if self.is_running:
            if callback:
                callback(f"{self.name} is already running")
            return False
        
        try:
            # Setup
            if not self.setup():
                if callback:
                    callback(f"Failed to setup {self.name}")
                return False
            
            self.is_running = True
            self.start_time = time.time()
            self.iteration_count = 0
            
            if callback:
                callback(f"Started {self.name} v{self.version}")
            
            # Main execution loop
            while self.is_running:
                self.iteration_count += 1
                
                # Execute one iteration
                if not self.execute_iteration():
                    break
                
                # Check duration
                elapsed = time.time() - self.start_time
                if elapsed >= duration:
                    if callback:
                        callback(f"{self.name} completed {self.iteration_count} iterations in {elapsed:.1f}s")
                    break
                
                # Small delay between iterations
                time.sleep(0.1)
            
            return True
            
        except Exception as e:
            if callback:
                callback(f"{self.name} error: {str(e)}")
            return False
        finally:
            self.cleanup()
            self.is_running = False
    
    def stop(self) -> None:
        """Stop the script execution"""
        self.is_running = False
    
    def get_status(self) -> Dict[str, Any]:
        """Get current script status"""
        elapsed = time.time() - self.start_time if self.start_time else 0
        
        return {
            'name': self.name,
            'is_running': self.is_running,
            'iteration_count': self.iteration_count,
            'elapsed_time': elapsed,
            'version': self.version
        }
    
    def log(self, message: str, callback: Optional[Callable] = None) -> None:
        """Log a message"""
        timestamp = time.strftime("%H:%M:%S")
        log_message = f"[{timestamp}] {self.name}: {message}"
        
        if callback:
            callback(log_message)
        else:
            print(log_message)
    
    def wait(self, duration: float) -> None:
        """Wait for specified duration"""
        time.sleep(duration)
    
    def wait_random(self, min_duration: float, max_duration: float) -> None:
        """Wait for random duration"""
        import random
        duration = random.uniform(min_duration, max_duration)
        time.sleep(duration)
    
    def get_elapsed_time(self) -> float:
        """Get elapsed execution time"""
        if self.start_time:
            return time.time() - self.start_time
        return 0.0
    
    def get_iteration_rate(self) -> float:
        """Get iterations per second"""
        elapsed = self.get_elapsed_time()
        if elapsed > 0:
            return self.iteration_count / elapsed
        return 0.0
