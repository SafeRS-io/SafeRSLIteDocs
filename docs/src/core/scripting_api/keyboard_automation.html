class KeyboardAutomation:
    """Handles keyboard-related automation functions"""
    
    def __init__(self, keyboard_controller: KeyboardController):
        self.keyboard_controller = keyboard_controller
    
    def press_key(self, key: Union[str, int], duration: float = 0.1) -> bool:
        """Press a key by key code or string name"""
        if not key:
            print("No key specified")
            return False
        
        # Convert string keys to key codes if needed
        key_code = self._get_key_code(key)
        if key_code is None:
            return False
        
        return self.keyboard_controller.press_key(key_code, duration)
    
    def send_string(self, text: str, delay: float = 0.05) -> bool:
        """Send a string of text"""
        if not text:
            print("No text specified")
            return False
        
        return self.keyboard_controller.send_string(text, delay)
    
    def press_key_combination(self, keys: List[Union[str, int]], delay: float = 0.1) -> bool:
        """Press a combination of keys (e.g., [VK.CTRL, VK.C])"""
        if not keys:
            print("No keys specified")
            return False
        
        try:
            # Press all keys down
            for key in keys:
                key_code = self._get_key_code(key)
                if key_code is not None:
                    self.keyboard_controller.key_down(key_code)
            
            # Hold for specified duration
            import time
            time.sleep(delay)
            
            # Release all keys in reverse order
            for key in reversed(keys):
                key_code = self._get_key_code(key)
                if key_code is not None:
                    self.keyboard_controller.key_up(key_code)
            
            return True
        except Exception as e:
            print(f"Key combination failed: {e}")
            return False
    
    def hold_key(self, key: Union[str, int]) -> bool:
        """Hold down a key"""
        if not key:
            print("No key specified")
            return False
        
        key_code = self._get_key_code(key)
        if key_code is None:
            return False
        
        return self.keyboard_controller.key_down(key_code)
    
    def release_key(self, key: Union[str, int]) -> bool:
        """Release a key"""
        if not key:
            print("No key specified")
            return False
        
        key_code = self._get_key_code(key)
        if key_code is None:
            return False
        
        return self.keyboard_controller.key_up(key_code)
    
    def type_text(self, text: str, delay: float = 0.05) -> bool:
        """Type text character by character (alias for send_string)"""
        return self.send_string(text, delay)
    
    def _get_key_code(self, key: Union[str, int]) -> Optional[int]:
        """Convert string key name to key code, or return int if already a key code"""
        if isinstance(key, int):
            return key
        
        # Convert string to key code
        key_map = {
            'space': VK.SPACE,
            'enter': VK.ENTER,
            'return': VK.ENTER,
            'escape': VK.ESCAPE,
            'esc': VK.ESCAPE,
            'tab': VK.TAB,
            'shift': VK.SHIFT,
            'ctrl': VK.CTRL,
            'control': VK.CTRL,
            'alt': VK.ALT,
            'menu': VK.ALT,
            
            # Function keys
            'f1': VK.F1, 'f2': VK.F2, 'f3': VK.F3, 'f4': VK.F4,
            'f5': VK.F5, 'f6': VK.F6, 'f7': VK.F7, 'f8': VK.F8,
            'f9': VK.F9, 'f10': VK.F10, 'f11': VK.F11, 'f12': VK.F12,
            
            # Arrow keys
            'up': VK.UP, 'down': VK.DOWN, 'left': VK.LEFT, 'right': VK.RIGHT,
            
            # Number keys
            '0': VK.NUM_0, '1': VK.NUM_1, '2': VK.NUM_2, '3': VK.NUM_3, '4': VK.NUM_4,
            '5': VK.NUM_5, '6': VK.NUM_6, '7': VK.NUM_7, '8': VK.NUM_8, '9': VK.NUM_9,
            
            # Letter keys
            'a': VK.A, 'b': VK.B, 'c': VK.C, 'd': VK.D, 'e': VK.E, 'f': VK.F,
            'g': VK.G, 'h': VK.H, 'i': VK.I, 'j': VK.J, 'k': VK.K, 'l': VK.L,
            'm': VK.M, 'n': VK.N, 'o': VK.O, 'p': VK.P, 'q': VK.Q, 'r': VK.R,
            's': VK.S, 't': VK.T, 'u': VK.U, 'v': VK.V, 'w': VK.W, 'x': VK.X,
            'y': VK.Y, 'z': VK.Z,
        }
        
        key_lower = key.lower()
        if key_lower in key_map:
            return key_map[key_lower]
        
        print(f"Unknown key: {key}")
        return None
