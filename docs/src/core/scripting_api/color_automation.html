class ColorAutomation:
    """Simple color detection and automation functions"""
    
    def __init__(self, color_scanner: ColorScanner, gear_config: AppConfig, mouse_controller):
        self.color_scanner = color_scanner
        self.gear_config = gear_config
        self.mouse_controller = mouse_controller
    
    def wait_for_color(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, 
                      timeout: float = 10.0, is_running_callback: Optional[Callable] = None, 
                      center_out: bool = False) -> bool:
        """Wait until a color appears in an area"""
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            if is_running_callback and not is_running_callback():
                return False
            
            if self.does_color_exist(area, color_hex, tolerance, center_out):
                return True
            
            time.sleep(0.1)
        
        return False
    
    def wait_til_no_color(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, 
                         timeout: tuple[float, float] = (2.0, 10.0)) -> bool:
        """
        Wait until a color is absent for a continuous period >= timeout[0] seconds,
        but give up entirely after timeout[1] seconds.
        """
        min_absent_time, max_total_time = timeout
        total_start = time.time()
        absent_start = None

        while time.time() - total_start < max_total_time:
            found = self._does_color_exist_pixel_scan(area, color_hex, tolerance)
            if not found:
                if absent_start is None:
                    absent_start = time.time()
                elif time.time() - absent_start >= min_absent_time:
                    return True
            else:
                absent_start = None  # Reset if color comes back
            time.sleep(0.1)

        return False
    
    def does_color_exist(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, 
                        center_out: bool = False) -> bool:
        """Check if a color exists in an area"""
        coordinates = self._find_color_coordinates(area, color_hex, tolerance, center_out)
        return len(coordinates) > 0
    
    def find_and_click_color(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, 
                            button: str = "left", center_out: bool = False) -> bool:
        """Find and click the center of a color in an area"""
        coordinates = self._find_color_coordinates(area, color_hex, tolerance, center_out)
        
        if coordinates:
            # Use the first (most prominent) coordinate
            x, y = coordinates[0]
            return self.mouse_controller.click(x, y, button)
        
        return False
    
    def get_color_at_position(self, x: int, y: int) -> Optional[Tuple[int, int, int]]:
        """Get RGB color at specific position"""
        return self.color_scanner.get_color_at_position(x, y)
    
    def find_color_in_area(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, 
                          center_out: bool = False) -> Optional[Tuple[int, int]]:
        """Find color in area and return first coordinate"""
        coordinates = self._find_color_coordinates(area, color_hex, tolerance, center_out)
        return coordinates[0] if coordinates else None
    
    def find_color_closest_to_center(self, area: Union[Dict, str], color_hex: str, 
                                   tolerance: int = 10) -> Optional[Tuple[int, int]]:
        """Find the color occurrence closest to the center of a circular area"""
        if not self._is_circular_area(area):
            return None
        
        area_bounds = self._get_area_bounds(area)
        color_rgb = self._hex_to_rgb(color_hex)
        
        if not area_bounds or not color_rgb:
            return None
        
        return self.color_scanner.find_color_closest_to_center(area_bounds, color_rgb, tolerance)
    
    def scan_orb_status(self, orb_area: Union[Dict, str], status_colors: Dict[str, str], 
                       tolerance: int = 10) -> Dict[str, bool]:
        """Scan an orb area for different status colors"""
        if not self._is_circular_area(orb_area):
            return {}
        
        area_bounds = self._get_area_bounds(orb_area)
        if not area_bounds:
            return {}
        
        # Convert hex colors to RGB
        rgb_colors = {}
        for status_name, color_hex in status_colors.items():
            color_rgb = self._hex_to_rgb(color_hex)
            if color_rgb:
                rgb_colors[status_name] = color_rgb
        
        return self.color_scanner.scan_orb_for_status(area_bounds, rgb_colors, tolerance)
    
    def get_all_colors_in_area(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, 
                              center_out: bool = False) -> List[Tuple[int, int]]:
        """Get all occurrences of a color in an area"""
        return self._find_color_coordinates(area, color_hex, tolerance, center_out)
    
    def get_all_colors_in_circle(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, 
                                center_out: bool = False) -> List[Tuple[int, int]]:
        """Get all occurrences of a color in a circular area (alias for get_all_colors_in_area)"""
        return self.get_all_colors_in_area(area, color_hex, tolerance, center_out)
    
    # Convenience methods for common operations
    def wait_for_xp_gain(self, xp_drop_area: Union[Dict, str], xp_color: str, 
                        tolerance: int = 10, timeout: float = 5.0) -> bool:
        """Wait for XP to stop being gained"""
        return self.wait_til_no_color(xp_drop_area, xp_color, tolerance, (10.0, 20.0))
    
    def wait_for_gained_xp(self, xp_drop_area: Union[Dict, str], xp_color: str, 
                          tolerance: int = 10, timeout: float = 10.0) -> bool:
        """Wait for XP drop area to be found - uses pixel scanning for thin text"""
        return self._wait_for_color_pixel_scan(xp_drop_area, xp_color, tolerance, timeout)
    
    # Private helper methods
    def _find_color_coordinates(self, area: Union[Dict, str], color_hex: str, tolerance: int, 
                               center_out: bool) -> List[Tuple[int, int]]:
        """Core method to find color coordinates - handles both rectangular and circular areas"""
        area_bounds = self._get_area_bounds(area)
        color_rgb = self._hex_to_rgb(color_hex)
        
        if not area_bounds or not color_rgb:
            return []
        
        # Check if this is a circular area
        if self._is_circular_area(area):
            if center_out:
                return self.color_scanner.scan_circular_area_center_out(area_bounds, color_rgb, tolerance)
            else:
                return self.color_scanner.scan_circular_area(area_bounds, color_rgb, tolerance)
        else:
            # Rectangular area - use smart detection
            return self.color_scanner.find_color_smart(color_rgb, tolerance, area_bounds)
    
    def _get_area_bounds(self, area: Union[Dict, str]) -> Optional[Union[Tuple[int, int, int, int], Tuple[int, int, int]]]:
        """Get area bounds from area object or string ID"""
        if isinstance(area, dict) and 'bounds' in area:
            return area['bounds']
        elif isinstance(area, str):
            return self.gear_config.get_scan_area_by_id(area)
        return None
    
    def _is_circular_area(self, area: Union[Dict, str]) -> bool:
        """Check if an area is circular"""
        if isinstance(area, str):
            return self.color_scanner.is_circular_area(area)
        elif isinstance(area, dict) and 'bounds' in area:
            return len(area['bounds']) == 3
        return False
    
    def _hex_to_rgb(self, color_hex: str) -> Optional[Tuple[int, int, int]]:
        """Convert hex color to RGB tuple"""
        try:
            if color_hex.startswith('#'):
                color_hex = color_hex[1:]
            return tuple(int(color_hex[i:i+2], 16) for i in (0, 2, 4))
        except (ValueError, IndexError):
            return None
    
    def _wait_for_color_pixel_scan(self, area: Union[Dict, str], color_hex: str, tolerance: int, timeout: float) -> bool:
        """Wait for color using pixel scanning (for thin text like XP)"""
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            if self._does_color_exist_pixel_scan(area, color_hex, tolerance):
                return True
            time.sleep(0.1)
        
        return False
    
    def _does_color_exist_pixel_scan(self, area: Union[Dict, str], color_hex: str, tolerance: int) -> bool:
        """Check if color exists using pixel scanning (for thin text like XP)"""
        area_bounds = self._get_area_bounds(area)
        color_rgb = self._hex_to_rgb(color_hex)
        
        if not area_bounds or not color_rgb:
            return False
        
        # Force pixel scanning for thin text
        coordinates = self.color_scanner.find_color_first(color_rgb, tolerance, area_bounds)
        return coordinates is not None
