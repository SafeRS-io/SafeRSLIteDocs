class ColorAutomation:
    """Handles color detection and related automation functions"""
    
    def __init__(self, color_scanner: ColorScanner, gear_config: AppConfig):
        self.color_scanner = color_scanner
        self.gear_config = gear_config
    
    def find_and_click_color(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, button: str = "left", center_out: bool = False) -> bool:
        """Find and click a color in an area - accepts area object or string ID"""
        # Get area bounds
        area_bounds = self._get_area_bounds(area)
        if not area_bounds:
            return False
        
        # Convert hex to RGB
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return False
        
        # Check if this is a circular area
        if self._is_circular_area(area):
            # Circular area: (center_x, center_y, radius)
            center_x, center_y, radius = area_bounds
            if center_out:
                coordinates = self.color_scanner.scan_circular_area_center_out(area_bounds, color_rgb, tolerance)
            else:
                coordinates = self.color_scanner.scan_circular_area(area_bounds, color_rgb, tolerance)
            
            if coordinates:
                x, y = coordinates[0]  # Use first match
                return self.color_scanner.mouse_controller.click(x, y, button)
            else:
                print(f"Color {color_hex} not found in circular area")
                return False
        else:
            # Rectangular area: (x1, y1, x2, y2)
            x1, y1, x2, y2 = area_bounds
            width = x2 - x1
            height = y2 - y1
            
            # Search for color in rectangular area
            result = self.color_scanner.find_color_coordinates(color_rgb, tolerance, area_bounds)
            
            if result:
                x, y = result[0]  # Use first match
                return self.color_scanner.mouse_controller.click(x, y, button)
            else:
                print(f"Color {color_hex} not found in rectangular area")
                return False
    
    def wait_for_color(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, timeout: float = 10.0, is_running_callback: Optional[callable] = None, center_out: bool = False) -> bool:
        """Wait until a color appears in an area - accepts area object or string ID"""
        # Get area bounds
        area_bounds = self._get_area_bounds(area)
        if not area_bounds:
            return False
        
        # Convert hex to RGB
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return False
        
        start_time = time.time()
        while time.time() - start_time < timeout:
            if is_running_callback and not is_running_callback():
                return False
            
            # Check if this is a circular area
            if self._is_circular_area(area):
                if center_out:
                    coordinates = self.color_scanner.scan_circular_area_center_out(area_bounds, color_rgb, tolerance)
                else:
                    coordinates = self.color_scanner.scan_circular_area(area_bounds, color_rgb, tolerance)
                result = len(coordinates) > 0
            else:
                # Rectangular area
                result = self.color_scanner.find_color_coordinates(color_rgb, tolerance, area_bounds)
                result = len(result) > 0 if result else False
            
            if result:
                return True
            
            time.sleep(0.1)
        
        print(f"Timeout waiting for color {color_hex} in area")
        return False

    def find_color_in_area(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, center_out: bool = False) -> Optional[Tuple[int, int]]:
        """Find color in area and return coordinates"""
        # Get area bounds
        area_bounds = self._get_area_bounds(area)
        if not area_bounds:
            return None
        
        # Convert hex to RGB
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return None
        
        # Check if this is a circular area
        if self._is_circular_area(area):
            if center_out:
                coordinates = self.color_scanner.scan_circular_area_center_out(area_bounds, color_rgb, tolerance)
            else:
                coordinates = self.color_scanner.scan_circular_area(area_bounds, color_rgb, tolerance)
            return coordinates[0] if coordinates else None
        else:
            # Rectangular area
            result = self.color_scanner.find_color_coordinates(color_rgb, tolerance, area_bounds)
            return result[0] if result else None
    
    # Circular area specific methods
    def find_color_closest_to_center(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10) -> Optional[Tuple[int, int]]:
        """Find the color occurrence closest to the center of a circular area"""
        if not self._is_circular_area(area):
            print(f"Area '{area}' is not a circular area")
            return None
        
        area_bounds = self._get_area_bounds(area)
        if not area_bounds:
            return None
        
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return None
        
        return self.color_scanner.find_color_closest_to_center(area_bounds, color_rgb, tolerance)
    
    def scan_orb_status(self, orb_area: Union[Dict, str], status_colors: Dict[str, str], tolerance: int = 10) -> Dict[str, bool]:
        """
        Scan an orb area for different status colors
        
        Args:
            orb_area: Orb area (health_orb, prayer_orb, run_orb, special_attack_orb)
            status_colors: Dictionary mapping status names to hex colors
            tolerance: Color tolerance for matching
            
        Returns:
            Dictionary mapping status names to boolean (True if color found)
        """
        if not self._is_circular_area(orb_area):
            print(f"Area '{orb_area}' is not a circular area")
            return {}
        
        area_bounds = self._get_area_bounds(orb_area)
        if not area_bounds:
            return {}
        
        # Convert hex colors to RGB
        rgb_colors = {}
        for status_name, color_hex in status_colors.items():
            color_rgb = self._hex_to_rgb(color_hex)
            if color_rgb:
                rgb_colors[status_name] = color_rgb
        
        return self.color_scanner.scan_orb_for_status(area_bounds, rgb_colors, tolerance)
    
    def get_all_colors_in_circle(self, area: Union[Dict, str], color_hex: str, tolerance: int = 10, center_out: bool = False) -> List[Tuple[int, int]]:
        """Get all occurrences of a color in a circular area"""
        if not self._is_circular_area(area):
            print(f"Area '{area}' is not a circular area")
            return []
        
        area_bounds = self._get_area_bounds(area)
        if not area_bounds:
            return []
        
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return []
        
        if center_out:
            return self.color_scanner.scan_circular_area_center_out(area_bounds, color_rgb, tolerance)
        else:
            return self.color_scanner.scan_circular_area(area_bounds, color_rgb, tolerance)
    
    def is_orb_low(self, orb_area: Union[Dict, str], low_color: str, tolerance: int = 10) -> bool:
        """Check if an orb is in low status (convenience method)"""
        return self.does_color_exist(orb_area, low_color, tolerance, center_out=True)
    
    def is_orb_high(self, orb_area: Union[Dict, str], high_color: str, tolerance: int = 10) -> bool:
        """Check if an orb is in high status (convenience method)"""
        return self.does_color_exist(orb_area, high_color, tolerance, center_out=True)
    
    def _get_area_bounds(self, area: Union[Dict, str]) -> Optional[Union[Tuple[int, int, int, int], Tuple[int, int, int]]]:
        """Get area bounds from area object or string ID - returns rectangular (x1,y1,x2,y2) or circular (center_x,center_y,radius)"""
        if isinstance(area, dict) and 'bounds' in area:
            # Area object passed directly
            return area['bounds']
        elif isinstance(area, str):
            # String ID passed - look up area
            area_data = self.gear_config.get_scan_area_by_id(area)
            if not area_data:
                print(f"Area '{area}' not found")
                return None
            return area_data
        else:
            print(f"Invalid area parameter: {area}")
            return None
    
    def _is_circular_area(self, area: Union[Dict, str]) -> bool:
        """Check if an area is circular based on its ID or data"""
        if isinstance(area, str):
            return self.color_scanner.is_circular_area(area)
        elif isinstance(area, dict) and 'bounds' in area:
            # Check if bounds data indicates circular area (3 values instead of 4)
            return len(area['bounds']) == 3
        return False
    
    def _hex_to_rgb(self, color_hex: str) -> Optional[Tuple[int, int, int]]:
        """Convert hex color to RGB tuple"""
        try:
            # Remove # if present
            if color_hex.startswith('#'):
                color_hex = color_hex[1:]
            
            # Convert to RGB
            return tuple(int(color_hex[i:i+2], 16) for i in (0, 2, 4))
        except (ValueError, IndexError):
            print(f"Invalid color format: {color_hex}")
            return None
