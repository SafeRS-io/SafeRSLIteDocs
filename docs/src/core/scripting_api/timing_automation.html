class TimingAutomation:
    """Handles timing-related automation functions"""
    
    def __init__(self):
        pass
    
    def wait(self, duration: float) -> bool:
        """Wait for specified duration"""
        if duration > 0:
            time.sleep(duration)
        return True
    
    def wait_random(self, min_duration: float, max_duration: float) -> bool:
        """Wait for random duration between min and max"""
        if min_duration < 0 or max_duration < min_duration:
            print("Invalid random wait parameters")
            return False
        
        duration = random.uniform(min_duration, max_duration)
        time.sleep(duration)
        return True
    
    def get_current_time(self) -> float:
        """Get current time in seconds since epoch"""
        return time.time()
    
    def get_elapsed_time(self, start_time: float) -> float:
        """Get elapsed time since start_time"""
        return time.time() - start_time
    
    def wait_until(self, condition_func, timeout: float = 10.0, check_interval: float = 0.1) -> bool:
        """
        Wait until a condition function returns True
        
        Args:
            condition_func: Function that returns True when condition is met
            timeout: Maximum time to wait in seconds
            check_interval: How often to check the condition in seconds
            
        Returns:
            bool: True if condition was met, False if timeout occurred
        """
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            try:
                if condition_func():
                    return True
            except Exception as e:
                print(f"Error in wait_until condition: {e}")
                return False
            
            time.sleep(check_interval)
        
        return False
    
    def wait_while(self, condition_func, timeout: float = 10.0, check_interval: float = 0.1) -> bool:
        """
        Wait while a condition function returns True
        
        Args:
            condition_func: Function that returns True while condition is met
            timeout: Maximum time to wait in seconds
            check_interval: How often to check the condition in seconds
            
        Returns:
            bool: True if condition stopped being met, False if timeout occurred
        """
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            try:
                if not condition_func():
                    return True
            except Exception as e:
                print(f"Error in wait_while condition: {e}")
                return False
            
            time.sleep(check_interval)
        
        return False
    
    def sleep_ms(self, milliseconds: int) -> bool:
        """
        Sleep for specified milliseconds
        
        Args:
            milliseconds: Number of milliseconds to sleep
            
        Returns:
            bool: Always True
        """
        if milliseconds > 0:
            time.sleep(milliseconds / 1000.0)
        return True
