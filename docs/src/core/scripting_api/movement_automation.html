class MovementAutomation:
    """Handles character movement detection and automation functions"""
    
    def __init__(self, color_scanner: ColorScanner, gear_config: AppConfig):
        self.color_scanner = color_scanner
        self.gear_config = gear_config
        self.last_minimap_state = None
        self.movement_threshold = 0.001  # 0.1% change threshold for movement detection
        self.minimap_region = None  # Will be set based on config
        
    def _get_minimap_region(self) -> Optional[Tuple[int, int, int, int]]:
        """Get minimap region from config or use default"""
        if self.minimap_region:
            return self.minimap_region
            
        # Try to get minimap area from config
        minimap_area = self.gear_config.get_scan_area_by_id('minimap')
        if minimap_area and len(minimap_area) == 4:
            # Ensure all coordinates are integers
            x1, y1, x2, y2 = minimap_area
            return (int(x1), int(y1), int(x2), int(y2))
        elif minimap_area and len(minimap_area) == 3:
            # Circular area - convert to rectangular bounds
            center_x, center_y, radius = minimap_area
            return (int(center_x - radius), int(center_y - radius), int(center_x + radius), int(center_y + radius))
        
        # Default minimap region (typical RuneScape minimap location)
        return (550, 10, 750, 210)
    
    def _capture_minimap_edges(self) -> Optional[np.ndarray]:
        """Capture the edges of the minimap for movement detection"""
        try:
            minimap_region = self._get_minimap_region()
            if not minimap_region:
                print("No minimap region found")
                return None
            
            # Capture a slightly larger region to ensure we get the edges
            x1, y1, x2, y2 = minimap_region
            padding = 5
            capture_region = (int(x1 - padding), int(y1 - padding), int(x2 + padding), int(y2 + padding))
            
            # Debug: Print the capture region to help identify issues
            print(f"Capturing minimap region: {capture_region}")
            
            screenshot = self.color_scanner.capture_window_screenshot(capture_region)
            if screenshot is None:
                print("Screenshot capture failed")
                return None
            
            # Enhanced edge detection to avoid false positives from other players
            height, width = screenshot.shape[:2]
            edge_thickness = 2
            
            # Create edge mask
            edge_mask = np.zeros((height, width), dtype=np.uint8)
            
            # Top edge
            edge_mask[:edge_thickness, :] = 255
            # Bottom edge  
            edge_mask[-edge_thickness:, :] = 255
            # Left edge
            edge_mask[:, :edge_thickness] = 255
            # Right edge
            edge_mask[:, -edge_thickness:] = 255
            
            # Also include center region where your character dot would be
            center_mask = np.zeros((height, width), dtype=np.uint8)
            center_x, center_y = width // 2, height // 2
            center_radius = min(width, height) // 6  # Small center region
            cv2.circle(center_mask, (center_x, center_y), center_radius, 255, -1)
            
            # Combine edge and center regions
            combined_mask = cv2.bitwise_or(edge_mask, center_mask)
            edges = cv2.bitwise_and(screenshot, screenshot, mask=combined_mask)
            
            # Debug: Print edge detection info
            print(f"Edge detection: shape={edges.shape}, non-zero pixels={np.count_nonzero(edges)}")
            
            # Return only the edge pixels to avoid false positives from other players
            return edges
            
        except Exception as e:
            print(f"Failed to capture minimap edges: {e}")
            # Try with default region as fallback
            try:
                default_region = (545, 5, 755, 215)  # Default with padding
                screenshot = self.color_scanner.capture_window_screenshot(default_region)
                if screenshot is None:
                    return None
                
                # Extract just the edge pixels
                height, width = screenshot.shape[:2]
                edge_thickness = 2
                
                # Create edge mask
                edge_mask = np.zeros((height, width), dtype=np.uint8)
                
                # Top edge
                edge_mask[:edge_thickness, :] = 255
                # Bottom edge  
                edge_mask[-edge_thickness:, :] = 255
                # Left edge
                edge_mask[:, :edge_thickness] = 255
                # Right edge
                edge_mask[:, -edge_thickness:] = 255
                
                # Apply mask to get only edge pixels
                edges = cv2.bitwise_and(screenshot, screenshot, mask=edge_mask)
                
                return edges
                
            except Exception as fallback_e:
                print(f"Fallback minimap capture also failed: {fallback_e}")
                return None
    
    def _calculate_movement_similarity(self, state1: np.ndarray, state2: np.ndarray) -> float:
        """Calculate similarity between two minimap edge states"""
        try:
            if state1 is None or state2 is None:
                return 0.0
            
            if state1.shape != state2.shape:
                return 0.0
            
            # Convert to grayscale for comparison
            gray1 = cv2.cvtColor(state1, cv2.COLOR_RGB2GRAY) if len(state1.shape) == 3 else state1
            gray2 = cv2.cvtColor(state2, cv2.COLOR_RGB2GRAY) if len(state2.shape) == 3 else state2
            
            # Calculate normalized cross-correlation as similarity measure
            gray1_norm = gray1.astype(np.float32) / 255.0
            gray2_norm = gray2.astype(np.float32) / 255.0
            
            # Calculate mean squared error
            mse = np.mean((gray1_norm - gray2_norm) ** 2)
            
            # Convert MSE to similarity (0 = completely different, 1 = identical)
            similarity = 1.0 - mse
            
            # Debug: Print similarity values for troubleshooting
            print(f"MSE: {mse:.6f}, Similarity: {similarity:.6f}")
            
            return max(0.0, min(1.0, similarity))  # Clamp to [0, 1]
            
        except Exception as e:
            print(f"Failed to calculate movement similarity: {e}")
            return 0.0
    
    def is_moving(self, threshold: Optional[float] = None) -> bool:
        """
        Check if the character is currently moving by comparing minimap edges
        
        Args:
            threshold: Movement threshold (0.0-1.0). Lower values = more sensitive.
                      Default uses self.movement_threshold
        
        Returns:
            True if character is moving, False if stationary
        """
        try:
            current_state = self._capture_minimap_edges()
            if current_state is None:
                return False
            
            if self.last_minimap_state is None:
                # First check - assume not moving
                self.last_minimap_state = current_state
                return False
            
            # Calculate similarity
            similarity = self._calculate_movement_similarity(self.last_minimap_state, current_state)
            
            # Update state for next check
            self.last_minimap_state = current_state
            
            # Use provided threshold or default
            movement_threshold = threshold if threshold is not None else self.movement_threshold
            
            # Lower similarity = more movement
            return similarity < (1.0 - movement_threshold)
            
        except Exception as e:
            print(f"Failed to check movement: {e}")
            return False
    
    def wait_until_moving(self, timeout: float = 10.0, 
                        threshold: Optional[float] = None,
                        is_running_callback: Optional[Callable] = None) -> bool:
        """
        Wait until the character starts moving
        
        Args:
            timeout: Maximum time to wait in seconds
            threshold: Movement threshold (0.0-1.0)
            is_running_callback: Optional callback to check if script should continue
            
        Returns:
            True if movement detected, False if timeout
        """
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            if is_running_callback and not is_running_callback():
                return False
            
            if self.is_moving(threshold):
                return True
            
            time.sleep(0.1)
        
        return False
    
    def wait_until_stationary(self, timeout: float = 10.0, 
                             threshold: Optional[float] = None,
                             is_running_callback: Optional[Callable] = None) -> bool:
        """
        Wait until the character stops moving
        
        Args:
            timeout: Maximum time to wait in seconds
            threshold: Movement threshold (0.0-1.0)
            is_running_callback: Optional callback to check if script should continue
            
        Returns:
            True if character stopped moving, False if timeout
        """
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            if is_running_callback and not is_running_callback():
                return False
            
            if not self.is_moving(threshold):
                return True
            
            time.sleep(0.1)
        
        return False
    
    def wait_for_movement_change(self, timeout: float = 10.0,
                                threshold: Optional[float] = None,
                                is_running_callback: Optional[Callable] = None) -> bool:
        """
        Wait for any movement change (start or stop moving)
        
        Args:
            timeout: Maximum time to wait in seconds
            threshold: Movement threshold (0.0-1.0)
            is_running_callback: Optional callback to check if script should continue
            
        Returns:
            True if movement state changed, False if timeout
        """
        start_time = time.time()
        initial_moving = self.is_moving(threshold)
        
        while time.time() - start_time < timeout:
            if is_running_callback and not is_running_callback():
                return False
            
            current_moving = self.is_moving(threshold)
            if current_moving != initial_moving:
                return True
            
            time.sleep(0.1)
        
        return False
    
    def get_movement_state(self, threshold: Optional[float] = None) -> Dict[str, Union[bool, float]]:
        """
        Get detailed movement state information
        
        Args:
            threshold: Movement threshold (0.0-1.0)
            
        Returns:
            Dictionary with movement state and similarity score
        """
        try:
            current_state = self._capture_minimap_edges()
            if current_state is None:
                return {"moving": False, "similarity": 0.0, "error": "Failed to capture minimap"}
            
            if self.last_minimap_state is None:
                self.last_minimap_state = current_state
                return {"moving": False, "similarity": 1.0, "note": "First check"}
            
            similarity = self._calculate_movement_similarity(self.last_minimap_state, current_state)
            movement_threshold = threshold if threshold is not None else self.movement_threshold
            
            self.last_minimap_state = current_state
            
            return {
                "moving": similarity < (1.0 - movement_threshold),
                "similarity": similarity,
                "threshold": movement_threshold
            }
            
        except Exception as e:
            return {"moving": False, "similarity": 0.0, "error": str(e)}
    
    def set_movement_threshold(self, threshold: float):
        """
        Set the movement detection threshold
        
        Args:
            threshold: Movement threshold (0.0-1.0). Lower values = more sensitive.
                      Recommended range: 0.02-0.1
        """
        if 0.0 <= threshold <= 1.0:
            self.movement_threshold = threshold
            print(f"Movement threshold set to: {threshold}")
        else:
            print("Movement threshold must be between 0.0 and 1.0")
    
    def set_minimap_region(self, x1: int, y1: int, x2: int, y2: int):
        """
        Set custom minimap region for movement detection
        
        Args:
            x1, y1: Top-left coordinates (relative to client window)
            x2, y2: Bottom-right coordinates (relative to client window)
        """
        self.minimap_region = (x1, y1, x2, y2)
        print(f"Minimap region set to: {self.minimap_region}")
    
    def reset_movement_state(self):
        """Reset the movement detection state (useful when changing areas)"""
        self.last_minimap_state = None
        print("Movement state reset")
