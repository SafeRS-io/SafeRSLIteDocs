class InventoryAutomation:
    """Simple inventory-related automation functions"""
    
    def __init__(self, color_scanner: ColorScanner, mouse_controller: MouseController, 
                 inventory_slots: List[Dict], keyboard_controller: KeyboardController):
        self.color_scanner = color_scanner
        self.mouse_controller = mouse_controller
        self.inventory_slots = inventory_slots
        self.keyboard_controller = keyboard_controller
        
    def click_slot(self, slot_id: int, button: str = "left") -> bool:
        """Click on an inventory slot with randomized position"""
        if not self._is_valid_slot(slot_id):
            return False
        
        slot_data = self.inventory_slots[slot_id]
        x, y = slot_data['center']
        
        # Add randomization within 3 pixel radius (slots are smaller than general areas)
        offset_x = random.randint(-3, 3)
        offset_y = random.randint(-3, 3)
        x += offset_x
        y += offset_y
        
        return self.mouse_controller.click(x, y, button)
    
    def drop_all_items(self) -> bool:
        """Drop all items in inventory"""
        # Ensure window is focused before sending shift key
        self._ensure_window_focus()
        
        # Try alternative shift key method first
        shift_success = self._send_shift_key_alternative()
        if not shift_success:
            # Fall back to original method
            self.keyboard_controller.key_down(VK.SHIFT)
        
        # Wait longer to ensure shift is properly registered before clicking
        time.sleep(random.uniform(0.3, 0.5))
        
        try:
            for slot_id in range(len(self.inventory_slots)):
                self.click_slot(slot_id, "left")
                time.sleep(random.uniform(0.03, 0.1))
        finally:
            # Release shift using the same method we used to press it
            if shift_success:
                self._release_shift_key_alternative()
            else:
                self.keyboard_controller.key_up(VK.SHIFT)
        
        return True
        
    def drop_all_color(self, color_hex: str, tolerance: int = 10) -> bool:
        """Drop all items with a specific color"""
        slots_to_drop = self._find_slots_with_color(color_hex, tolerance)
        
        if not slots_to_drop:
            return False
        
        # Ensure window is focused before sending shift key
        self._ensure_window_focus()
        
        # Try alternative shift key method first
        shift_success = self._send_shift_key_alternative()
        if not shift_success:
            # Fall back to original method
            self.keyboard_controller.key_down(VK.SHIFT)
        
        # Wait longer to ensure shift is properly registered before clicking
        time.sleep(random.uniform(0.3, 0.5))
        
        try:
            for slot_id in slots_to_drop:
                self.click_slot(slot_id, "left")
                time.sleep(random.uniform(0.1, 0.3))
        finally:
            # Release shift using the same method we used to press it
            if shift_success:
                self._release_shift_key_alternative()
            else:
                self.keyboard_controller.key_up(VK.SHIFT)
        
        return True
    
    def wait_for_color_at_slot(self, slot_id: int, color_hex: str, tolerance: int = 10, 
                              timeout: float = 10.0, is_running_callback: Optional[Callable] = None) -> bool:
        """Wait until a color appears in an inventory slot"""
        if not self._is_valid_slot(slot_id):
            return False
        
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return False
        
        start_time = time.time()
        while time.time() - start_time < timeout:
            if is_running_callback and not is_running_callback():
                return False
            
            if self._check_color_in_slot(slot_id, color_rgb, tolerance):
                return True
            
            time.sleep(0.1)
        
        return False
    
    def does_color_exist_at_slot(self, slot_id: int, color_hex: str, tolerance: int = 10) -> bool:
        """Check if a color exists in an inventory slot"""
        if not self._is_valid_slot(slot_id):
            return False
        
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return False
        
        return self._check_color_in_slot(slot_id, color_rgb, tolerance)
    
    def find_color_in_slot(self, slot_id: int, color_hex: str, tolerance: int = 10) -> Optional[Tuple[int, int]]:
        """Find color in specific slot and return coordinates"""
        if not self._is_valid_slot(slot_id):
            return None
        
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return None
        
        return self._find_color_coordinates_in_slot(slot_id, color_rgb, tolerance)
    
    def get_slot_info(self, slot_id: int) -> Optional[Dict]:
        """Get information about a specific inventory slot"""
        if not self._is_valid_slot(slot_id):
            return None
        
        return self.inventory_slots[slot_id].copy()
    
    def get_slot_count(self) -> int:
        """Get total number of inventory slots"""
        return len(self.inventory_slots)
    
    def update_inventory_slots(self, inventory_slots: List[Dict]):
        """Update inventory slots reference"""
        self.inventory_slots = inventory_slots
    
    # Private helper methods
    def _is_valid_slot(self, slot_id: int) -> bool:
        """Check if slot ID is valid"""
        return 0 <= slot_id < len(self.inventory_slots)
    
    def _get_slot_bounds(self, slot_id: int) -> Optional[Tuple[int, int, int, int]]:
        """Get bounds for a slot - inventory slots should always have bounds"""
        slot_data = self.inventory_slots[slot_id]
        
        # Inventory slots should always have bounds from the scan area
        if 'bounds' in slot_data:
            return slot_data['bounds']
        
        # Fallback: if no bounds, something is wrong
        print(f"WARNING: Slot {slot_id} has no bounds! Slot data: {slot_data}")
        return None
    
    def _check_color_in_slot(self, slot_id: int, color_rgb: Tuple[int, int, int], tolerance: int) -> bool:
        """Check if color exists in a slot"""
        bounds = self._get_slot_bounds(slot_id)
        if not bounds:
            return False
        
        # Use smart detection that automatically chooses the best method
        coordinates = self.color_scanner.find_color_smart(color_rgb, tolerance, bounds)
        found = len(coordinates) > 0
        
        # Debug first few slots
        if slot_id < 3:
            print(f"DEBUG: Slot {slot_id} bounds: {bounds}, found: {found}, coordinates: {len(coordinates)}")
        
        return found
    
    def _find_color_coordinates_in_slot(self, slot_id: int, color_rgb: Tuple[int, int, int], tolerance: int) -> Optional[Tuple[int, int]]:
        """Find color coordinates in a slot"""
        bounds = self._get_slot_bounds(slot_id)
        if not bounds:
            return None
        
        # Use smart detection that automatically chooses the best method
        coordinates = self.color_scanner.find_color_smart(color_rgb, tolerance, bounds)
        
        if coordinates:
            # Calculate center of all found coordinates
            if len(coordinates) == 1:
                return coordinates[0]
            else:
                # Calculate average position (center of all found pixels)
                avg_x = sum(coord[0] for coord in coordinates) // len(coordinates)
                avg_y = sum(coord[1] for coord in coordinates) // len(coordinates)
                return (avg_x, avg_y)
        
        return None
    
    def _find_slots_with_color(self, color_hex: str, tolerance: int) -> List[int]:
        """Find all slots that contain a specific color"""
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return []
        
        # Debug: Check what's in the inventory slots
        print(f"DEBUG: Checking {len(self.inventory_slots)} inventory slots")
        if len(self.inventory_slots) > 0:
            print(f"DEBUG: First slot data: {self.inventory_slots[0]}")
        
        slots_with_color = []
        for slot_id in range(len(self.inventory_slots)):
            if self._check_color_in_slot(slot_id, color_rgb, tolerance):
                slots_with_color.append(slot_id)
        
        print(f"DEBUG: Found {len(slots_with_color)} slots with color {color_hex}: {slots_with_color}")
        return slots_with_color
    
    def _hex_to_rgb(self, color_hex: str) -> Optional[Tuple[int, int, int]]:
        """Convert hex color to RGB tuple"""
        try:
            if color_hex.startswith('#'):
                color_hex = color_hex[1:]
            return tuple(int(color_hex[i:i+2], 16) for i in (0, 2, 4))
        except (ValueError, IndexError):
            return None
    
    def _ensure_window_focus(self):
        """Ensure the game window is focused before sending keyboard input"""
        try:
            import win32gui
            import win32con
            
            # Get the window handle from the mouse controller's window manager
            if hasattr(self.mouse_controller, 'window_manager') and self.mouse_controller.window_manager.is_attached():
                handle = self.mouse_controller.window_manager.attached_handle
                
                # Bring window to foreground
                win32gui.SetForegroundWindow(handle)
                
                # Send a brief activation message
                win32gui.PostMessage(handle, win32con.WM_ACTIVATE, win32con.WA_ACTIVE, 0)
                
                # Small delay to ensure focus is established
                time.sleep(0.1)
                
        except Exception as e:
            print(f"Warning: Could not ensure window focus: {e}")
    
    def _send_shift_key_alternative(self):
        """Alternative method to send shift key using different approaches"""
        try:
            import win32gui
            import win32con
            import win32api
            
            # Get the window handle
            if hasattr(self.mouse_controller, 'window_manager') and self.mouse_controller.window_manager.is_attached():
                handle = self.mouse_controller.window_manager.attached_handle
                
                # Method 1: Try using SendInput API (more direct)
                try:
                    import ctypes
                    from ctypes import wintypes
                    
                    # Define input structure
                    class POINT(ctypes.Structure):
                        _fields_ = [("x", ctypes.c_long), ("y", ctypes.c_long)]
                    
                    class MOUSEINPUT(ctypes.Structure):
                        _fields_ = [("dx", ctypes.c_long),
                                   ("dy", ctypes.c_long),
                                   ("mouseData", wintypes.DWORD),
                                   ("dwFlags", wintypes.DWORD),
                                   ("time", wintypes.DWORD),
                                   ("dwExtraInfo", ctypes.POINTER(ctypes.c_ulong))]
                    
                    class KEYBDINPUT(ctypes.Structure):
                        _fields_ = [("wVk", wintypes.WORD),
                                   ("wScan", wintypes.WORD),
                                   ("dwFlags", wintypes.DWORD),
                                   ("time", wintypes.DWORD),
                                   ("dwExtraInfo", ctypes.POINTER(ctypes.c_ulong))]
                    
                    class HARDWAREINPUT(ctypes.Structure):
                        _fields_ = [("uMsg", wintypes.DWORD),
                                   ("wParamL", wintypes.WORD),
                                   ("wParamH", wintypes.WORD)]
                    
                    class INPUT(ctypes.Structure):
                        class _INPUT(ctypes.Union):
                            _fields_ = [("ki", KEYBDINPUT),
                                       ("mi", MOUSEINPUT),
                                       ("hi", HARDWAREINPUT)]
                        _anonymous_ = ("_input",)
                        _fields_ = [("type", wintypes.DWORD),
                                   ("_input", _INPUT)]
                    
                    # Create input for shift key down
                    input_down = INPUT()
                    input_down.type = 1  # INPUT_KEYBOARD
                    input_down.ki.wVk = win32con.VK_SHIFT
                    input_down.ki.dwFlags = 0  # Key down
                    
                    # Send the input
                    ctypes.windll.user32.SendInput(1, ctypes.byref(input_down), ctypes.sizeof(INPUT))
                    return True
                    
                except Exception as e:
                    print(f"SendInput method failed: {e}")
                
                # Method 2: Try using keybd_event (older but sometimes more reliable)
                try:
                    win32api.keybd_event(win32con.VK_SHIFT, 0, 0, 0)  # Key down
                    return True
                except Exception as e:
                    print(f"keybd_event method failed: {e}")
                
                # Method 3: Fall back to PostMessage with different parameters
                try:
                    # Try with different lParam values
                    win32gui.PostMessage(handle, win32con.WM_KEYDOWN, win32con.VK_SHIFT, 0x001D0001)
                    return True
                except Exception as e:
                    print(f"PostMessage method failed: {e}")
                
        except Exception as e:
            print(f"All shift key methods failed: {e}")
            return False
        
        return False
    
    def _release_shift_key_alternative(self):

                                   ("wParamL", wintypes.WORD),

                                   ("_input", _INPUT)]
                    
                    # Create input for shift key up
                    input_up = INPUT()
                    input_up.type = 1  # INPUT_KEYBOARD
                    input_up.ki.wVk = win32con.VK_SHIFT
                    input_up.ki.dwFlags = 2  # KEYEVENTF_KEYUP

                    return True
                    
                except Exception as e:

                # Method 2: Try using keybd_event
                try:
                    win32api.keybd_event(win32con.VK_SHIFT, 0, 2, 0)  # Key up (2 = KEYEVENTF_KEYUP)
                    return True
                except Exception as e:
                    print(f"keybd_event release method failed: {e}")
                
                # Method 3: Fall back to PostMessage
                try:
                    win32gui.PostMessage(handle, win32con.WM_KEYUP, win32con.VK_SHIFT, 0xC01D0001)
                    return True
                except Exception as e:
                    print(f"PostMessage release method failed: {e}")
                
        except Exception as e:
            print(f"All shift release methods failed: {e}")
            return False
        
        return False
