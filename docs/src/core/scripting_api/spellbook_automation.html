class SpellbookAutomation:
    """Handles spellbook-related automation functions"""
    
    def __init__(self, color_scanner: ColorScanner, mouse_controller: MouseController, spellbook_slots: List[Dict]):
        self.color_scanner = color_scanner
        self.mouse_controller = mouse_controller
        self.spellbook_slots = spellbook_slots
        
    def cast_spell(self, spell_name: str) -> bool:
        """Cast a spell. Spell names are mapped to their respective slot ids and then clicked."""
        # map spell names to slot ids then use click_spell to cast
        spell_map = {
            "Blind": 0,
            "Crumble Undead": 1,
            "High Alch": 55,
            "High Level Alchemy": 55,
            "Low Alch": 23,
            "Low Level Alchemy": 23,
            "Teleblock": 27,
            "Enchant": 31,
        }
        
        pass
    
    def click_spell(self, slot_id: int, button: str = "left") -> bool:
        """Click on a spellbook slot with randomized position"""
        if slot_id < 0 or slot_id >= len(self.spellbook_slots):
            print(f"Invalid spell slot ID: {slot_id}")
            return False
        
        slot_data = self.spellbook_slots[slot_id]
        x, y = slot_data['center']
        
        # Add randomization within 3 pixel radius (spell slots are similar size to inventory slots)
        offset_x = random.randint(-3, 3)
        offset_y = random.randint(-3, 3)
        x += offset_x
        y += offset_y
        
        return self.mouse_controller.click(x, y, button)
    
    def wait_for_color_at_spell(self, slot_id: int, color_hex: str, tolerance: int = 10, timeout: float = 10.0, is_running_callback: Optional[callable] = None) -> bool:
        """Wait until a color appears in a spellbook slot"""
        if slot_id < 0 or slot_id >= len(self.spellbook_slots):
            print(f"Invalid spell slot ID: {slot_id}")
            return False
        
        # Convert hex to RGB
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return False
        
        slot_data = self.spellbook_slots[slot_id]
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            if is_running_callback and not is_running_callback():
                return False
            
            result = self.color_scanner.find_color_in_area(
                slot_data['x'], slot_data['y'], slot_data['width'], slot_data['height'],
                color_rgb, tolerance
            )
            
            if result:
                return True
            
            time.sleep(0.1)
        
        print(f"Timeout waiting for color {color_hex} in spell slot {slot_id}")
        return False
    
    def does_color_exist_at_spell(self, slot_id: int, color_hex: str, tolerance: int = 10) -> bool:
        """Check if a color exists in a spellbook slot"""
        if slot_id < 0 or slot_id >= len(self.spellbook_slots):
            print(f"Invalid spell slot ID: {slot_id}")
            return False
        
        # Convert hex to RGB
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return False
        
        slot_data = self.spellbook_slots[slot_id]
        
        # Search for color in slot
        result = self.color_scanner.find_color_in_area(
            slot_data['x'], slot_data['y'], slot_data['width'], slot_data['height'],
            color_rgb, tolerance
        )
        return result is not None
    
    def get_spell_info(self, slot_id: int) -> Optional[Dict]:
        """Get information about a specific spellbook slot"""
        if slot_id < 0 or slot_id >= len(self.spellbook_slots):
            print(f"Invalid spell slot ID: {slot_id}")
            return None
        
        return self.spellbook_slots[slot_id].copy()
    
    def get_spell_count(self) -> int:
        """Get total number of spellbook slots"""
        return len(self.spellbook_slots)
    
    def find_color_in_spell(self, slot_id: int, color_hex: str, tolerance: int = 10) -> Optional[Tuple[int, int]]:
        """Find color in specific spell slot and return coordinates"""
        if slot_id < 0 or slot_id >= len(self.spellbook_slots):
            print(f"Invalid spell slot ID: {slot_id}")
            return None
        
        # Convert hex to RGB
        color_rgb = self._hex_to_rgb(color_hex)
        if not color_rgb:
            return None
        
        slot_data = self.spellbook_slots[slot_id]
        
        return self.color_scanner.find_color_in_area(
            slot_data['x'], slot_data['y'], slot_data['width'], slot_data['height'],
            color_rgb, tolerance
        )
    
    def update_spellbook_slots(self, spellbook_slots: List[Dict]):
        """Update spellbook slots reference"""
        self.spellbook_slots = spellbook_slots
    
    def _hex_to_rgb(self, color_hex: str) -> Optional[Tuple[int, int, int]]:
        """Convert hex color to RGB tuple"""
        try:
            # Remove # if present
            if color_hex.startswith('#'):
                color_hex = color_hex[1:]
            
            # Convert to RGB
            return tuple(int(color_hex[i:i+2], 16) for i in (0, 2, 4))
        except (ValueError, IndexError):
            print(f"Invalid color format: {color_hex}")
            return None
