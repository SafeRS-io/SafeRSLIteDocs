class HotkeyAutomation:
    """Handles hotkey registration and management for scripts"""
    
    def __init__(self):
        self.script_hotkeys = {}  # Dictionary to store script-registered hotkeys
        self.hotkeys_enabled = False
        self.hotkey_thread = None
    
    def register_hotkey(self, hotkey: str, callback: Callable, description: str = "") -> bool:
        """
        Register a hotkey for the current script
        
        Args:
            hotkey: Hotkey string (e.g., "ctrl+1", "f1", "alt+q")
            callback: Function to call when hotkey is pressed
            description: Description of what the hotkey does
            
        Returns:
            bool: True if successful, False otherwise
        """
        try:
            if hotkey in self.script_hotkeys:
                self.unregister_hotkey(hotkey)
            
            self.script_hotkeys[hotkey] = {
                'callback': callback,
                'description': description,
                'registered': False
            }
            
            if self.hotkeys_enabled:
                self._register_single_hotkey(hotkey)
            
            print(f"HotkeyAutomation: Registered script hotkey {hotkey}")
            return True
            
        except Exception as e:
            print(f"Failed to register script hotkey {hotkey}: {e}")
            return False
    
    def unregister_hotkey(self, hotkey: str) -> bool:
        """
        Unregister a script hotkey
        
        Args:
            hotkey: Hotkey string to unregister
            
        Returns:
            bool: True if successful, False otherwise
        """
        try:
            if hotkey in self.script_hotkeys:
                if self.script_hotkeys[hotkey]['registered']:
                    keyboard.remove_hotkey(hotkey)
                del self.script_hotkeys[hotkey]
                print(f"HotkeyAutomation: Unregistered script hotkey {hotkey}")
            return True
            
        except Exception as e:
            print(f"Failed to unregister script hotkey {hotkey}: {e}")
            return False
    
    def start_hotkey_monitoring(self):
        """Start hotkey monitoring for script hotkeys"""
        if self.hotkeys_enabled:
            return
        
        self.hotkeys_enabled = True
        self.hotkey_thread = threading.Thread(target=self._hotkey_monitor, daemon=True)
        self.hotkey_thread.start()
        print("HotkeyAutomation: Hotkey monitoring started")
    
    def stop_hotkey_monitoring(self):
        """Stop hotkey monitoring and clear all script hotkeys"""
        if not self.hotkeys_enabled:
            return
        
        self.hotkeys_enabled = False
        
        # Unregister all script hotkeys
        for hotkey in list(self.script_hotkeys.keys()):
            self.unregister_hotkey(hotkey)
        
        if self.hotkey_thread:
            self.hotkey_thread.join(timeout=1)
        
        print("HotkeyAutomation: Hotkey monitoring stopped")
    
    def get_registered_hotkeys(self) -> Dict[str, Dict]:
        """Get all currently registered hotkeys"""
        return self.script_hotkeys.copy()
    
    def is_hotkey_registered(self, hotkey: str) -> bool:
        """Check if a hotkey is registered"""
        return hotkey in self.script_hotkeys
    
    def _register_single_hotkey(self, hotkey: str):
        """Register a single script hotkey"""
        try:
            if hotkey in self.script_hotkeys and not self.script_hotkeys[hotkey]['registered']:
                keyboard.add_hotkey(hotkey, self.script_hotkeys[hotkey]['callback'], suppress=True)
                self.script_hotkeys[hotkey]['registered'] = True
        except Exception as e:
            print(f"Failed to register script hotkey {hotkey}: {e}")
    
    def _hotkey_monitor(self):
        """Monitor thread for script hotkeys"""
        while self.hotkeys_enabled:
            try:
                time.sleep(0.1)  # Small delay to prevent high CPU usage
            except:
                break
