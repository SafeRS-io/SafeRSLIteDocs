class ScriptingEngine:
    """Core scripting engine for automation tasks"""
    
    def __init__(self, gear_config: AppConfig, color_scanner: ColorScanner, 
                 mouse_controller: MouseController, keyboard_controller: KeyboardController, 
                 inventory_slots: List[Dict], spellbook_slots: List[Dict] = None):
        self.gear_config = gear_config
        self.color_scanner = color_scanner
        self.mouse_controller = mouse_controller
        self.keyboard_controller = keyboard_controller
        self.inventory_slots = inventory_slots
        self.spellbook_slots = spellbook_slots or []
        
        # Initialize automation modules
        self.mouse_automation = MouseAutomation(mouse_controller, gear_config)
        self.color_automation = ColorAutomation(color_scanner, gear_config)
        self.keyboard_automation = KeyboardAutomation(keyboard_controller)
        self.timing_automation = TimingAutomation()
        self.hotkey_automation = HotkeyAutomation()
        self.inventory_automation = InventoryAutomation(color_scanner, mouse_controller, inventory_slots)
        self.spellbook_automation = SpellbookAutomation(color_scanner, mouse_controller, spellbook_slots)
        
        # Script execution state
        self.is_running = False
        self.current_script = None
        self.execution_thread = None
    
    def execute_base_script(self, script_instance: BaseScript, duration: float, 
                           callback: Optional[Callable] = None, log_callback: Optional[Callable] = None) -> bool:
        """
        Execute a BaseScript instance
        
        Args:
            script_instance: BaseScript instance to execute
            duration: Duration in seconds to run the script
            callback: Optional callback for status updates
            log_callback: Optional callback for log messages
            
        Returns:
            bool: True if execution started successfully
        """
        if self.is_running:
            if callback:
                callback("Script already running")
            return False
        
        if not isinstance(script_instance, BaseScript):
            if callback:
                callback("Invalid script instance - must be BaseScript")
            return False
        
        if log_callback:
            log_callback(f"Injecting automation modules into {script_instance.name}...")
        
        # Inject automation modules into the script instance
        self._inject_automation_modules(script_instance)
        
        if log_callback:
            log_callback(f"Automation modules injected successfully")
        
        self.is_running = True
        self.current_script = script_instance
        
        if log_callback:
            log_callback(f"Starting BaseScript execution thread...")
        
        # Run in separate thread to avoid blocking UI
        self.execution_thread = threading.Thread(
            target=self._run_base_script, 
            args=(script_instance, duration, callback, log_callback),
            daemon=True
        )
        self.execution_thread.start()
        
        if callback:
            callback(f"BaseScript '{script_instance.name}' started for {duration}s")
        
        return True
    
    def _run_base_script(self, script_instance: BaseScript, duration: float, callback: Optional[Callable], log_callback: Optional[Callable]):
        """Run the BaseScript execution"""
        try:
            # Override the script's log method to use our callback
            original_log = script_instance.log
            def custom_log(message: str):
                if log_callback:
                    log_callback(message)
                else:
                    original_log(message)
            script_instance.log = custom_log
            
            # Override the script's run method to check our is_running flag
            original_run = script_instance.run
            def custom_run(duration: float, callback: Optional[Callable] = None) -> bool:
                """Custom run method that checks ScriptingEngine's is_running flag"""
                if script_instance.is_running:
                    if callback:
                        callback(f"{script_instance.name} is already running")
                    return False
                
                try:
                    # Setup
                    if not script_instance.setup():
                        if callback:
                            callback(f"Failed to setup {script_instance.name}")
                        return False
                    
                    script_instance.is_running = True
                    script_instance.start_time = time.time()
                    script_instance.iteration_count = 0
                    
                    if callback:
                        callback(f"Started {script_instance.name} v{script_instance.version}")
                    
                    # Main execution loop - check both flags
                    while script_instance.is_running and self.is_running:
                        script_instance.iteration_count += 1
                        
                        # Execute one iteration
                        if not script_instance.execute_iteration():
                            break
                        
                        # Check duration
                        elapsed = time.time() - script_instance.start_time
                        if elapsed >= duration:
                            if callback:
                                callback(f"{script_instance.name} completed {script_instance.iteration_count} iterations in {elapsed:.1f}s")
                            break
                        
                        # Small delay between iterations
                        time.sleep(0.1)
                    
                    return True
                    
                except Exception as e:
                    if callback:
                        callback(f"{script_instance.name} error: {str(e)}")
                    return False
                finally:
                    script_instance.cleanup()
                    script_instance.is_running = False
            
            # Replace the run method temporarily
            script_instance.run = custom_run
            
            # Run the script
            success = script_instance.run(duration, callback)
            
            if success and callback:
                callback(f"BaseScript '{script_instance.name}' completed successfully")
            elif not success and callback:
                callback(f"BaseScript '{script_instance.name}' failed or stopped")
                
        except Exception as e:
            if callback:
                callback(f"BaseScript execution error: {str(e)}")
        finally:
            self.is_running = False
            self.current_script = None
    
    def _inject_automation_modules(self, script_instance: BaseScript):
        """Inject automation modules into a BaseScript instance"""
        # Add automation modules as attributes to the script instance
        script_instance.mouse = self.mouse_automation
        script_instance.color = self.color_automation
        script_instance.keyboard = self.keyboard_automation
        script_instance.timing = self.timing_automation
        script_instance.hotkey = self.hotkey_automation
        script_instance.inventory = self.inventory_automation
        script_instance.spellbook = self.spellbook_automation
        
        # Add convenience methods
        script_instance.click_slot = self.inventory_automation.click_slot
        script_instance.click_spell = self.spellbook_automation.click_spell
        script_instance.find_color = self.color_automation.does_color_exist
        script_instance.wait_for_color = lambda area, color_hex, tolerance=10, timeout=10.0: self.color_automation.wait_for_color(area, color_hex, tolerance, timeout, lambda: self.is_running)
        
        # Add scan area variables
        scan_areas = self.gear_config.get_all_scan_areas()
        for area_id, area_data in scan_areas.items():
            if area_data:
                if len(area_data) == 3:
                    # Circular area
                    center_x, center_y, radius = area_data
                    setattr(script_instance, area_id, {
                        'center_x': center_x,
                        'center_y': center_y,
                        'radius': radius,
                        'center': (center_x, center_y),
                        'bounds': area_data
                    })
                else:
                    # Rectangular area
                    x1, y1, x2, y2 = area_data
                    setattr(script_instance, area_id, {
                        'x': x1,
                        'y': y1,
                        'width': x2 - x1,
                        'height': y2 - y1,
                        'center': ((x1 + x2) // 2, (y1 + y2) // 2),
                        'bounds': area_data
                    })
            else:
                setattr(script_instance, area_id, None)
    
    def execute_script(self, python_code: str, duration: float, loop_enabled: bool, 
                       callback: Optional[Callable] = None, log_callback: Optional[Callable] = None) -> bool:
        """
        Execute Python code with available automation functions
        
        Args:
            python_code: Python code string to execute
            duration: Duration in seconds to run the script in a loop
            loop_enabled: Whether to run in a loop (always True for UI compatibility)
            callback: Optional callback for status updates
            log_callback: Optional callback for log messages (print output)
            
        Returns:
            bool: True if execution started successfully
        """
        if self.is_running:
            if callback:
                callback("Script already running")
            return False
        
        if not python_code.strip():
            if callback:
                callback("No code to execute")
            return False
        
        # Basic code safety check
        if not self._is_code_safe(python_code):
            if callback:
                callback("Code contains potentially dangerous operations")
            return False
        
        self.is_running = True
        self.current_script = python_code
        
        # Run in separate thread to avoid blocking UI
        self.execution_thread = threading.Thread(
            target=self._run_script, 
            args=(python_code, duration, callback, log_callback),
            daemon=True
        )
        self.execution_thread.start()
        
        if callback:
            callback(f"Script started in loop mode for {duration}s")
        
        return True
    
    def _run_script(self, python_code: str, duration: float, callback: Optional[Callable], log_callback: Optional[Callable]):
        """Run the script execution loop"""
        try:
            start_time = time.time()
            iteration = 0
            
            # Create the execution environment with available functions
            exec_globals = self._create_execution_environment(log_callback)
            
            while self.is_running:
                iteration += 1
                
                # Execute the Python code
                exec(python_code, exec_globals)
                
                # Check if we've exceeded the loop duration
                elapsed_time = time.time() - start_time
                if elapsed_time >= duration:
                    if callback:
                        callback(f"Script completed {iteration} iterations in {elapsed_time:.1f}s")
                    break
                
                # Small delay between iterations when looping
                if self.is_running:
                    time.sleep(0.1)
            
            if self.is_running:  # Only show success if not manually stopped
                elapsed_time = time.time() - start_time
                if callback:
                    callback(f"Script completed {iteration} iterations in {elapsed_time:.1f}s")
                    
        except Exception as e:
            if callback:
                callback(f"Script execution error: {str(e)}")
        finally:
            self.is_running = False
            self.current_script = None
    
    def _is_code_safe(self, code: str) -> bool:
        """Basic check for obviously dangerous code"""
        dangerous = ['eval(', 'exec(', '__import__', 'open(']
        return not any(danger in code for danger in dangerous)
    
    def stop_script(self):
        """Stop the currently running script"""
        self.is_running = False
        
        # If we have a BaseScript instance, stop it too
        if isinstance(self.current_script, BaseScript):
            self.current_script.stop()
        
        # Stop hotkey monitoring when script stops
        self.hotkey_automation.stop_hotkey_monitoring()
        
        # Wait for thread to finish with timeout
        if self.execution_thread and self.execution_thread.is_alive():
            self.execution_thread.join(timeout=2.0)
    
    def start_hotkey_monitoring(self):
        """Start hotkey monitoring for script hotkeys"""
        self.hotkey_automation.start_hotkey_monitoring()
    
    def stop_hotkey_monitoring(self):
        """Stop hotkey monitoring and clear all script hotkeys"""
        self.hotkey_automation.stop_hotkey_monitoring()
    
    def register_script_hotkey(self, hotkey: str, callback: Callable, description: str = "") -> bool:
        """
        Register a hotkey for the current script
        
        Args:
            hotkey: Hotkey string (e.g., "ctrl+1", "f1", "alt+q")
            callback: Function to call when hotkey is pressed
            description: Description of what the hotkey does
            
        Returns:
            bool: True if successful, False otherwise
        """
        return self.hotkey_automation.register_hotkey(hotkey, callback, description)
    
    def unregister_script_hotkey(self, hotkey: str) -> bool:
        """
        Unregister a script hotkey
        
        Args:
            hotkey: Hotkey string to unregister
            
        Returns:
            bool: True if successful, False otherwise
        """
        return self.hotkey_automation.unregister_hotkey(hotkey)
    
    def _create_execution_environment(self, log_callback: Optional[Callable] = None) -> Dict[str, Any]:
        """Create the execution environment with available automation functions"""
        # Create custom print function that captures output
        def custom_print(*args, **kwargs):
            """Custom print function that sends output to log callback"""
            import sys
            from io import StringIO
            
            # Capture the print output
            old_stdout = sys.stdout
            sys.stdout = captured_output = StringIO()
            
            try:
                # Call the original print function
                print(*args, **kwargs)
                output = captured_output.getvalue()
                
                # Send to log callback if available
                if log_callback and output.strip():
                    log_callback(output.strip())
                    
            finally:
                # Restore stdout
                sys.stdout = old_stdout
        
        # Get available scan areas and create global variables
        scan_areas = self.gear_config.get_all_scan_areas()
        
        # Get custom areas
        custom_areas = self.gear_config.get_custom_areas()
        
        # Create global variables for scan areas
        area_vars = {}
        for area_id, area_data in scan_areas.items():
            if area_data:
                # Check if this is a circular area (3 coordinates) or rectangular (4 coordinates)
                if len(area_data) == 3:
                    # Circular area: (center_x, center_y, radius)
                    center_x, center_y, radius = area_data
                    area_vars[area_id] = {
                        'center_x': center_x,
                        'center_y': center_y,
                        'radius': radius,
                        'center': (center_x, center_y),
                        'bounds': area_data
                    }
                else:
                    # Rectangular area: (x1, y1, x2, y2)
                    x1, y1, x2, y2 = area_data
                    area_vars[area_id] = {
                        'x': x1,
                        'y': y1,
                        'width': x2 - x1,
                        'height': y2 - y1,
                        'center': ((x1 + x2) // 2, (y1 + y2) // 2),
                        'bounds': area_data
                    }
            else:
                area_vars[area_id] = None
        
        # Add custom areas to global variables
        for area_name, (description, area_type) in custom_areas.items():
            if area_name not in area_vars:  # Don't override existing areas
                area_vars[area_name] = {
                    'description': description,
                    'bounds': None,  # Will be set when area is configured
                    'configured': False
                }
        
        return {
            # Mouse functions
            'click_area': self.mouse_automation.click_area,
            'click_coordinates': self.mouse_automation.click_coordinates,
            'move': self.mouse_automation.move,
            
            # Inventory functions
            'click_slot': self.inventory_automation.click_slot,
            'get_slot_info': self.inventory_automation.get_slot_info,
            'get_slot_count': self.inventory_automation.get_slot_count,
            
            # Spellbook functions
            'click_spell': self.spellbook_automation.click_spell,
            'get_spell_info': self.spellbook_automation.get_spell_info,
            'get_spell_count': self.spellbook_automation.get_spell_count,
            
            # Wait functions
            'wait': self.timing_automation.wait,
            'wait_random': self.timing_automation.wait_random,
            'wait_until': self.timing_automation.wait_until,
            'wait_while': self.timing_automation.wait_while,
            'sleep_ms': self.timing_automation.sleep_ms,
            
            # Color detection functions
            'find_and_click_color': self.color_automation.find_and_click_color,
            'wait_for_color': lambda area, color_hex, tolerance=10, timeout=10.0: 
                self.color_automation.wait_for_color(area, color_hex, tolerance, timeout, lambda: self.is_running),
            'wait_for_color_at_slot': lambda slot_id, color_hex, tolerance=10, timeout=10.0: 
                self.inventory_automation.wait_for_color_at_slot(slot_id, color_hex, tolerance, timeout, lambda: self.is_running),
            'wait_for_color_at_spell': lambda slot_id, color_hex, tolerance=10, timeout=10.0: 
                self.spellbook_automation.wait_for_color_at_spell(slot_id, color_hex, tolerance, timeout, lambda: self.is_running),
            'does_color_exist': self.color_automation.does_color_exist,
            'does_color_exist_at_slot': self.inventory_automation.does_color_exist_at_slot,
            'does_color_exist_at_spell': self.spellbook_automation.does_color_exist_at_spell,
            'get_color_at_position': self.color_automation.get_color_at_position,

    def get_available_areas(self) -> Dict[str, Optional[Dict]]:
        """Get available areas for compatibility with old UI code"""
        # This method is kept for compatibility with old UI code
        # In the new Python-based system, users just reference area IDs directly
        areas = {}
        if hasattr(self.gear_config, 'get_all_scan_areas'):
            all_areas = self.gear_config.get_all_scan_areas()
            for area_id, area_data in all_areas.items():
                if area_data:
                    # Convert tuple to dict format expected by old UI
                    x1, y1, x2, y2 = area_data
                    areas[area_id] = {
                        'x': x1,
                        'y': y1,
                        'width': x2 - x1,
                        'height': y2 - y1,
                        'center': ((x1 + x2) // 2, (y1 + y2) // 2)
                    }
                else:
                    areas[area_id] = None
        return areas

def stop_script(self):
    """Stop the currently running script"""
    self.is_running = False
    
    # If we have a BaseScript instance, stop it too
    if isinstance(self.current_script, BaseScript):
        self.current_script.stop()
    
    # Stop hotkey monitoring when script stops
    self.hotkey_automation.stop_hotkey_monitoring()
    
    # Wait for thread to finish with timeout
    if self.execution_thread and self.execution_thread.is_alive():
        self.execution_thread.join(timeout=2.0)
